const SHEET_NAME = "Sheet1"; // 1. HANDLE REGISTRATIONS function doPost(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); var data = JSON.parse(e.postData.contents); // Security Check if (data.auth_id !== "ceosafeparkindia" || data.auth_pass !== "Original@007") { return ContentService.createTextOutput(JSON.stringify({status: 'error', message: 'Unauthorized'})) .setMimeType(ContentService.MimeType.JSON); } // ** FIX: Use the ID sent from the App, or create one if missing ** var carId = data.id || "SP" + Utilities.formatDate(new Date(), "IST", "ddHHmmss"); // Save to Sheet // Columns: A=ID, B=Time, C=Name, D=Phone, E=Email, F=Vehicle, G=EmgName, H=EmgPhone, I=ShowNumber sheet.appendRow([ carId, new Date(), data.name, data.phone, data.email, data.vehicle, data.emg_name, data.emg_phone, data.show_number ]); return ContentService.createTextOutput(JSON.stringify({status: 'success', id: carId})) .setMimeType(ContentService.MimeType.JSON); } catch(error) { return ContentService.createTextOutput(JSON.stringify({status: 'error', message: error.toString()})) .setMimeType(ContentService.MimeType.JSON); } } // 2. HANDLE SCANS & MESSAGES (No changes needed here, but keeping it for completeness) function doGet(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); var carId = e.parameter.id; var action = e.parameter.action; var data = sheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { if (data[i][0] == carId) { var ownerData = { name: data[i][2], phone: data[i][3], email: data[i][4], vehicle: data[i][5], emg_name: data[i][6], emg_phone: data[i][7], show_number: data[i][8] }; if (action === 'message' && e.parameter.msg) { var userMsg = e.parameter.msg; if (ownerData.email) { MailApp.sendEmail({ to: ownerData.email, subject: "🚗 SafePark Alert: Someone contacted you!", htmlBody: "Hello " + ownerData.name + ",

" + "Someone scanned your QR Code (" + ownerData.vehicle + ").

" + "Message:
" + "
" + userMsg + "
" + "
Team SafePark India" }); } return ContentService.createTextOutput(JSON.stringify({status: 'success', message: 'Message Sent!'})) .setMimeType(ContentService.MimeType.JSON); } var responseJson = { status: 'success', data: { name: ownerData.name, vehicle: ownerData.vehicle, emg_name: ownerData.emg_name, emg_phone: ownerData.emg_phone, phone: (ownerData.show_number === "Yes") ? ownerData.phone : null } }; return ContentService.createTextOutput(JSON.stringify(responseJson)) .setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput(JSON.stringify({status: 'error', message: 'ID Not Found'})) .setMimeType(ContentService.MimeType.JSON); } catch(error) { return ContentService.createTextOutput(JSON.stringify({status: 'error', message: error.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
